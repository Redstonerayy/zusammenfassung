    class Car:
        def __init__(self, name: str, speed: float):
            self.name = name      # Attribut
            self.speed = speed
            self.__km_driven = 0
            
        def getDriven() -> float: # Methode
            return self.__km_driven
            
        def drive(hours: float):
            self.__km_driven += self.speed * hours
    \end{lstlisting}
